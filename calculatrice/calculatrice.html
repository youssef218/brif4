Enonce

Créez 4 fonctions Javascript (Somme, Soustraction, Division, Multiplication). L’utilisateur saisie deux nombres dans deux
inputs, choisi ensuite une opération arithmétique. Le résultat est donné par l’une des 4 fonctions Javascript.

document.getElementById('mul').onclick=function(){ let num1=document.getElementById('nombr1').value; let num2=document.getElementById('nombr2').value; document.getElementById('result').innerHTML=num1*num2 } document.getElementById('dev').onclick=function(){ let num1=document.getElementById('nombr1').value; let num2=document.getElementById('nombr2').value; document.getElementById('result').innerHTML=num1/num2 } document.getElementById('moin').onclick=function(){ let num1=document.getElementById('nombr1').value; let num2=document.getElementById('nombr2').value; document.getElementById('result').innerHTML=num1-num2 } document.getElementById('plus').onclick=function(){ let num1=document.getElementById('nombr1').value; let num2=document.getElementById('nombr2').value; num1=Number(num1) ;num2=Number(num2); document.getElementById('result').innerHTML=num1+num2 }
